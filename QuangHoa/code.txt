#include<math.h>
//chân ST_CP c?a 74HC595
int latchPin = 8;
//chân SH_CP c?a 74HC595
int clockPin = 12;
//Chân DS c?a 74HC595
int dataPin = 11;
 
int btn1=2;
int btn2=3;
int btn3=4;

//Tr?ng thái c?a LED, hay chính là byte mà ta s? g?i qua shiftOut
byte ledStatus;
void setup() {
  //B?n BU?C PH?I pinMode các chân này là OUTPUT
  Serial.begin(9600);
  pinMode(latchPin, OUTPUT);
  pinMode(clockPin, OUTPUT);
  pinMode(dataPin, OUTPUT);
  pinMode(btn1, INPUT);
  pinMode(btn2,INPUT);
}

void loop() { 
  //Serial.println(digitalRead(btn2)); 
  if (digitalRead(btn1)==1)
    phaitrai();
  if(digitalRead(btn2)==1)
    traiphai();
  delay(200);
}
void phaitrai()
{
  
  ledStatus = 0;
  for (int i = 0; i < 8; i++) {
    ledStatus = (ledStatus << 1) | 1; 
    
    
    
    digitalWrite(latchPin, LOW); //các dèn LED s? không sáng khi b?n digital LOW
 
    //ShiftOut ra IC
    shiftOut(dataPin, clockPin, MSBFIRST, ledStatus);  

    digitalWrite(latchPin, HIGH);//các dèn LED s? sáng v?i tr?ng thái v?a du?c c?p nh?p
   
    /**
      K?t thúc b?t bu?c ph?i có
    **/
    
    delay(500); // D?ng chuong trình kho?ng 500 mili giây d? th?y các hi?u ?ng c?a dèn LED
  }
  
  //T?t tu?n t?
  for (int i = 0;i<8;i++) {
    ledStatus <<= 1; //Ð?y t?t c? các bit qua bên trái 1 bit
    digitalWrite(latchPin, LOW);
    shiftOut(dataPin, clockPin, MSBFIRST, ledStatus);  
    digitalWrite(latchPin, HIGH);
    delay(500);
  }
}
void traiphai()
{
  //Sáng tu?n t? t? trái sang ph?i
  ledStatus = 0;
  int x1 = 0;
  
  for (int i = 7; i >=0; i--) 
  {
    x1 = pow(2,i) + 1;
    Serial.println(x1);
    ledStatus = ledStatus|x1;// L?i trình biên d?ch
    
    Serial.println(ledStatus);
    digitalWrite(latchPin, LOW);
    
    shiftOut(dataPin, clockPin, MSBFIRST, ledStatus);  
    
    digitalWrite(latchPin, HIGH);
    delay(500); 
  }
  
  //T?t tu?n t?
  for (int i = 7 ;i >= 0; i--)
  {
    x1 = pow(2,i)-1;
   // ledStatus = x1;
    digitalWrite(latchPin, LOW);
    shiftOut(dataPin, clockPin, MSBFIRST, x1);  
    digitalWrite(latchPin, HIGH);
    delay(500);
  }
}
